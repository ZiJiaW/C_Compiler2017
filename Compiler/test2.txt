int a[100];
int gcd(int m, int n)
{
    if(m==0)
        return (n);
    if(n==0)
        return (m);
    if(m>=n)
        return (gcd(m-n,n));
    else
        return (gcd(m,n-m));
}
int compare(int a,int b){
    if(a>b) return (1);
    if(a<=b) return (-1);
}
void bubblesort(int n)
{
    int k, i, tmp;
    for (i = n - 1; i >= 1; i = i - 1)
    {
        for (k = 0; k <= i - 1; k = k + 1)
        {
            if (a[k]>a[k + 1])
            {
                tmp = a[k];
                a[k] = a[k + 1];
                a[k + 1] = tmp;
            }
        }
    }
}
int fib(int n){
    if(n==0) return(0);
    if(n==1) return(1);
    return (fib(n-1)+fib(n-2));
}
